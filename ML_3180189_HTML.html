    # convert an array of values into a dataset matrix
    def create_dataset(dataset):
      dataX, dataY = [], []
      for i in range(len(dataset)-1):
        dataX.append(dataset[i])
        dataY.append(dataset[i + 1])
      return np.asarray(dataX), np.asarray(dataY)
